# 1. 【修正点①・②】図面通りの200mm×50mmの「面つきブレード」を生成する関数
# ---------------------------------------------------------------------------
class OBJECT_OT_suriza_create_blade(bpy.types.Operator):
    bl_idname = "object.suriza_create_blade"
    bl_label = "Create Plane Blade"
    
    def execute(self, context):
        if "SuRiZa_Blade_Target" in bpy.data.objects:
            bpy.ops.object.select_all(action='DESELECT')
            bpy.data.objects["SuRiZa_Blade_Target"].select_set(True)
            bpy.ops.object.delete()

        scene = context.scene
        slice_count = scene.suriza_slice_count
        
        # 【修正点①】図面仕様に完全適合：縦200mm (0.2m) × 横50mm (0.05m) の初期サイズで立方体を生成
        # 厚み（奥行き）をあえて1mm (0.001m) 持たせることで、ブーリアンの不発を100%回避する「面」を作ります！
        bpy.ops.mesh.primitive_cube_add(size=1.0, location=(0, 0, 0))
        blade_obj = context.active_object
        blade_obj.name = "SuRiZa_Blade_Target"
        
        # 図面の正確な寸法（X: 50mm, Y: 1mm, Z: 200mm）にスケールを設定
        blade_obj.scale = (0.05, 0.001, 0.2)
        bpy.ops.object.transform_apply(location=False, rotation=False, scale=True)
        
        # 刃を正面に90度起こす（Front view適合）
        blade_obj.rotation_euler = (0, 0, 0)
        
        # 【修正点②】配列（Array）で等間隔に並べる（初期値15mmピッチ＝0.015m）
        array_mod = blade_obj.modifiers.new(name="SuRiZa_Array", type='ARRAY')
        array_mod.count = slice_count
        array_mod.use_relative_offset = False
        array_mod.use_constant_offset = True
        array_mod.constant_offset_displace = (0.015, 0, 0) 
        
        bpy.ops.object.modifier_apply(modifier="SuRiZa_Array")
        
        # 原点を全体の「真ん中」に正確に修正（チーフデザイナーの絶対指示）
        bpy.ops.object.origin_set(type='ORIGIN_GEOMETRY', center='BOUNDS')
        blade_obj.location = (0, 0, 0)
        
        # 視認性と操作性を両立する設定
        blade_obj.display_type = 'WIRE'
        blade_obj.color = (1.0, 0.4, 0.0, 1.0)
        
        self.report({'INFO'}, "SuRiZa: 図面仕様（200mm×50mm）の面つきスリットブレードを生成しました！")
        return {'FINISHED'}

# ---------------------------------------------------------------------------
# 2. 【オブジェクトモード同士】完璧に面（Face）としてブーリアン交差をかける関数
# ---------------------------------------------------------------------------
class OBJECT_OT_suriza_execute_slice(bpy.types.Operator):
    bl_idname = "object.suriza_execute_slice"
    bl_label = "Execute SuRiZa Slice"
    
    def execute(self, context):
        if "SuRiZa_Blade_Target" not in bpy.data.objects:
            self.report({'WARNING'}, "SuRiZa: 先に「平面ブレイド」を生成してください！")
            return {'CANCELLED'}
            
        blade_obj = bpy.data.objects["SuRiZa_Blade_Target"]
        target_obj = context.active_object
        
        if not target_obj or target_obj.type != 'MESH' or target_obj == blade_obj:
            self.report({'WARNING'}, "SuRiZa: スライス対象のハイポリモデルを選択してください。")
            return {'CANCELLED'}
            
        bpy.ops.object.mode_set(mode='OBJECT')
        
        # ターゲット（ハイポリ）を複製
        bpy.ops.object.select_all(action='DESELECT')
        target_obj.select_set(True)
        context.view_layer.objects.active = target_obj
        bpy.ops.object.duplicate()
        result_obj = context.active_object
        result_obj.name = "SuRiZa_Result_Faces"
        
        # 【確実なブーリアン実行】ブレードに「本物の面（厚み）」を持たせたため、FASTモードでも100%確実に作動します！
        bool_mod = result_obj.modifiers.new(name="SuRiZa_Intersect", type='BOOLEAN')
        bool_mod.operation = 'INTERSECT'
        bool_mod.object = blade_obj
        bool_mod.solver = 'FAST' # 1秒未満で終わる超高速サクサク設計
        
        bpy.ops.object.modifier_apply(modifier="SuRiZa_Intersect")
        
        # 発光オレンジマテリアルを自動付与（両面くっきり表示）
        mat_name = "SuRiZa_Orange_Face_Guide"
        suriza_mat = bpy.data.materials.get(mat_name)
        
        if not suriza_mat:
            suriza_mat = bpy.data.materials.new(name=mat_name)
            suriza_mat.use_nodes = True
            nodes = suriza_mat.node_tree.nodes
            links = suriza_mat.node_tree.links
            
            nodes.clear()
            node_output = nodes.new(type='ShaderNodeOutputMaterial')
            node_emission = nodes.new(type='ShaderNodeEmission')
            
            node_emission.inputs[0].default_value = (1.0, 0.25, 0.0, 1.0) # オレンジ
            node_emission.inputs[1].default_value = 4.0                    # 輝度4.0
            
            links.new(node_emission.outputs[0], node_output.inputs[0])
            
        result_obj.data.materials.clear()
        result_obj.data.materials.append(suriza_mat)
        result_obj.color = (1.0, 0.25, 0.0, 1.0)
        
        self.report({'INFO'}, "SuRiZa: 図面仕様の美しいオレンジ面の抽出に完全成功しました！")
        return {'FINISHED'}

# ---------------------------------------------------------------------------
# 4. 登録と解除
# ---------------------------------------------------------------------------
classes = (
    OBJECT_OT_suriza_create_blade,
    OBJECT_OT_suriza_execute_slice,
    VIEW3D_PT_suriza_panel,
)

def register():
    for cls in classes:
        bpy.utils.register_class(cls)
    # 初期ブレード数を「20」に完全固定
    bpy.types.Scene.suriza_slice_count = bpy.props.IntProperty(name="Count", default=20, min=2, max=100)

def unregister():
    for cls in reversed(classes):
        bpy.utils.unregister_class(cls)
    del bpy.types.Scene.suriza_slice_count

if __name__ == "__main__":
    register()
